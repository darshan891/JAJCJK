from gensim.models import Word2Vec
import random

def main():
    corpus = [
        "AI is revolutionizing many industries including healthcare and finance",
        "Deep learning models are improving NLP applications",
        "Machine learning is widely used in predictive analytics",
        "Neural networks are at the core of modern AI",
        "Gen AI models create realistic text and images"
    ]

    data = [line.lower().split() for line in corpus]

    model = Word2Vec(
        data,
        vector_size=100,
        window=3,
        min_count=1
    )

    prompt = "How does AI impact the future of healthcare?"

    ai = random.choice(model.wv.most_similar("ai", topn=3))[0]
    health = random.choice(model.wv.most_similar("healthcare", topn=3))[0]

    print("Original Prompt:", prompt)
    print("Enriched Prompt:")
    print(f"How does {ai} impact the future of {health}?")

if __name__ == "__main__":
    main()
