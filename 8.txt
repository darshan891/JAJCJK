import cohere
from google.colab import drive

def main():
    drive.mount("/content/drive")

    with open(
        "/content/drive/MyDrive/article.txt",
        "r",
        encoding="utf-8"
    ) as file:
        text = file.read()

    co = cohere.Client("YOUR_COHERE_API_KEY")

    response = co.chat(
        model="command-a-03-2025",
        message=f"""
Analyze this document:
{text}

Give:
1. Keywords
2. Summary (3 sentences)
3. Main Topics
4. Named Entities (People, Organizations, Locations, Dates)

Format:
Keywords:
Summary:
Topics:
Entities:
"""
    )

    print(response.text)

if __name__ == "__main__":
    main()
