import cohere
import wikipediaapi

def main():
    co = cohere.Client("YOUR_COHERE_API_KEY")

    wiki = wikipediaapi.Wikipedia(
        user_agent="MyApp",
        language="en"
    )

    institution = input("Enter Institution Name: ")
    page = wiki.page(institution)

    if page.exists():
        response = co.chat(
            model="command-a-03-2025",
            message=f"""
Extract:
1. Founder
2. Founded Year
3. Branches
4. Employees
5. Summary (4 lines)

Text:
{page.summary[:2000]}
"""
        )

        print(response.text)

    else:
        print("Institution not found!")

if __name__ == "__main__":
    main()
