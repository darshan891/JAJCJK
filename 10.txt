from pypdf import PdfReader

def main():
    reader = PdfReader("sample.pdf")

    text = ""

    for page in reader.pages:
        text += page.extract_text()

    print("PDF Chatbot Ready! Type 'exit' to stop.")

    while True:
        q = input("You: ")

        if q.lower() == "exit":
            break

        if q.lower() in text.lower():
            print("Bot: Information found in the PDF.")
        else:
            print("Bot: Sorry, answer not found in the PDF.")

if __name__ == "__main__":
    main()
