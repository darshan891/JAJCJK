from transformers import pipeline

def main():
    summarizer = pipeline(
        "summarization",
        model="facebook/bart-large-cnn"
    )

    text = """
    Artificial Intelligence (AI) is transforming the way we interact with technology.
    Generative AI creates new content such as text, images, and code.
    It uses deep learning to generate original outputs and is used in healthcare,
    education, business, and many other industries.
    """

    summary = summarizer(
        text,
        max_length=50,
        min_length=20,
        do_sample=False
    )

    print("Original Text Length:", len(text.split()))
    print("Summary:", summary[0]["summary_text"])

if __name__ == "__main__":
    main()
