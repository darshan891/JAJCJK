from gensim.models import Word2Vec

def main():
    corpus = [
        "The patient was diagnosed with diabetes and prescribed metformin.",
        "CT scan revealed signs of pneumonia in the left lungs.",
        "Doctors recommend a balanced diet to manage hypertension.",
        "Symptoms of COVID-19 include fever cough and shortness of breath.",
        "The new cancer treatment involves immunotherapy."
    ]

    data = [line.lower().split() for line in corpus]

    model = Word2Vec(
        data,
        vector_size=100,
        window=3,
        min_count=1
    )

    print("Custom Word2Vec Model Trained Successfully!")

    print("\nWords similar to 'diabetes':")
    print(model.wv.most_similar("diabetes"))

    print("\nWords similar to 'scan':")
    print(model.wv.most_similar("scan"))

    print("\nVector for 'diabetes':")
    print(model.wv["diabetes"])

if __name__ == "__main__":
    main()
