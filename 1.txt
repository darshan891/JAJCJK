import gensim.downloader as api

def main():
    model = api.load("word2vec-google-news-300")

    print("Result of King - Man + Woman:")
    print(model.most_similar(
        positive=["king", "woman"],
        negative=["man"],
        topn=5
    ))

    print("\nWords similar to 'computer':")
    print(model.most_similar(
        "computer",
        topn=5
    ))

if __name__ == "__main__":
    main()
